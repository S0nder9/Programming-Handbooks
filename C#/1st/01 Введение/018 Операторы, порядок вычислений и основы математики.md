```csharp
int num1 = 5;
int num2 = 13;

Console.WriteLine("123" + num1 - num2);
// Ошибка: сначала выполняется "123" + num1 → получается строка "1235".
// Затем C# пытается выполнить "1235" - 13, а вычитать число из строки нельзя.

Console.WriteLine("123b" + (num1 - num2));
// Сначала выполняются скобки: 5 - 13 = -8
// Затем строка "123b" объединяется с результатом → "123b-8"

Console.WriteLine("123b" + num1 * num2);
// Сначала умножение (*), так как у него приоритет выше, чем у +
// 5 * 13 = 65
// Потом конкатенация: "123b" + 65 → "123b65"

Console.WriteLine("123b" + (num1 * num2));
// То же самое: "123b65"

Console.WriteLine("123b" + num1 / num2);
// Сначала деление: 5 / 13 = 0
// Почему 0? Потому что int / int = целочисленное деление (остаток отбрасывается)
// Потом: "123b" + 0 → "123b0"

Console.WriteLine("123b" + (num1 / num2));
// То же самое: "123b0"

Console.WriteLine("123b" + num1 % num2);
// Остаток от деления: 5 % 13 = 5
// Потом: "123b" + 5 → "123b5"

Console.WriteLine("123b" + (num1 % num2));
// То же самое: "123b5"

Console.ReadKey();
```
