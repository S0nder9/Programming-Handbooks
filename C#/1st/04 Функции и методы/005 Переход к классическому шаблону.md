## Переход к классическому шаблону

```csharp
namespace OldStyle // Используется для упорядовачивания кода (как папки). Название прастравнства имен
{
internal class Program // Название файла программы, доступен в рамках 1 сборки
{
    static void Main(string[] args) // static значит, что метод не зависит от конкретного экземпляра класса, а принадлежит именно этому.
    {
        Console.WriteLine("Hello, World!");
    }
}

}
```
